## Usage Examples

### Python: Fetch and Display Report

```python
import httpx
import asyncio
from datetime import datetime

async def get_daily_report(date: str):
    """Fetch and display daily report."""
    async with httpx.AsyncClient() as client:
        response = await client.get(
            f"http://localhost:8000/api/v1/reports/daily/{date}"
        )
        response.raise_for_status()
        return response.json()

async def main():
    # Get report for specific date
    report = await get_daily_report("2024-01-15")
    
    # Display executive summary
    print("Executive Summary:")
    print(report["executive_summary"])
    print()
    
    # Display projects
    print("Projects:")
    for project in report["projects"]:
        print(f"- {project['repository']}: {project['summary']}")
    print()
    
    # Display contributors
    print("Contributors:")
    for contributor in report["contributors"]:
        print(f"- {contributor['name']}: {contributor['summary']}")

if __name__ == "__main__":
    asyncio.run(main())
```

### JavaScript: Fetch Latest Report

```javascript
async function getLatestReport() {
  const response = await fetch(
    'http://localhost:8000/api/v1/reports/daily/latest'
  );
  
  if (!response.ok) {
    throw new Error(`HTTP error! status: ${response.status}`);
  }
  
  const report = await response.json();
  
  // Display report
  console.log('Executive Summary:');
  console.log(report.executive_summary);
  console.log();
  
  console.log('Projects:');
  report.projects.forEach(project => {
    console.log(`- ${project.repository}: ${project.summary}`);
  });
  
  return report;
}

// Usage
getLatestReport()
  .then(report => console.log('Report fetched successfully'))
  .catch(error => console.error('Error:', error));
```

## Testing with Sample Data

### Load Sample Data into Database

```python
import asyncio
import json
from sqlalchemy.ext.asyncio import AsyncSession
from app.models import Commit, Repository
from app.database import get_db

async def load_sample_commits():
    """Load sample commits into database."""
    with open('docs/examples/sample-commits.json') as f:
        commits_data = json.load(f)
    
    async with get_db() as session:
        for commit_data in commits_data:
            # Create or get repository
            repo = await session.execute(
                select(Repository).where(
                    Repository.name == commit_data['repository']
                )
            )
            repo = repo.scalar_one_or_none()
            
            if not repo:
                repo = Repository(
                    name=commit_data['repository'],
                    full_name=f"acme/{commit_data['repository']}",
                    url=f"https://git.acme.com/acme/{commit_data['repository']}"
                )
                session.add(repo)
                await session.flush()
            
            # Create commit
            commit = Commit(
                repository_id=repo.id,
                sha=commit_data['sha'],
                author_name=commit_data['author_name'],
                author_email=commit_data['author_email'],
                timestamp=commit_data['timestamp'],
                title=commit_data['title'],
                description=commit_data['description'],
                files_changed=commit_data['files_changed'],
                insertions=commit_data['insertions'],
                deletions=commit_data['deletions']
            )
            session.add(commit)
        
        await session.commit()
        print(f"Loaded {len(commits_data)} sample commits")

if __name__ == "__main__":
    asyncio.run(load_sample_commits())
```

## Integration Examples

### Email Report Delivery

```python
import smtplib
from email.mime.text import MIMEText
from email.mime.multipart import MIMEMultipart

async def send_daily_report_email(report, recipient):
    """Send daily report via email."""
    # Create message
    msg = MIMEMultipart('alternative')
    msg['Subject'] = f"Daily Engineering Report - {report['report_date']}"
    msg['From'] = "reports@cogence.example.com"
    msg['To'] = recipient
    
    # Create HTML content
    html = f"""
    <html>
      <body>
        <h1>Daily Engineering Report</h1>
        <h2>Executive Summary</h2>
        <p>{report['executive_summary']}</p>
        
        <h2>Projects</h2>
        <ul>
          {''.join(f"<li><strong>{p['repository']}</strong>: {p['summary']}</li>" 
                   for p in report['projects'])}
        </ul>
        
        <h2>Contributors</h2>
        <ul>
          {''.join(f"<li><strong>{c['name']}</strong>: {c['summary']}</li>" 
                   for c in report['contributors'])}
        </ul>
        
        <h2>Management Notes</h2>
        <p>{report['management_notes']}</p>
      </body>
    </html>
    """
    
    # Attach HTML
    msg.attach(MIMEText(html, 'html'))
    
    # Send email
    with smtplib.SMTP('localhost') as server:
        server.send_message(msg)
    
    print(f"Report sent to {recipient}")
```

### Slack Integration

```python
import httpx

async def post_report_to_slack(report, webhook_url):
    """Post daily report to Slack."""
    # Format message
    message = {
        "blocks": [
            {
                "type": "header",
                "text": {
                    "type": "plain_text",
                    "text": f"Daily Engineering Report - {report['report_date']}"
                }
            },
            {
                "type": "section",
                "text": {
                    "type": "mrkdwn",
                    "text": f"*Executive Summary*\n{report['executive_summary']}"
                }
            },
            {
                "type": "section",
                "text": {
                    "type": "mrkdwn",
                    "text": "*Projects*\n" + "\n".join(
                        f"• *{p['repository']}*: {p['summary']}" 
                        for p in report['projects']
                    )
                }
            }
        ]
    }
    
    # Post to Slack
    async with httpx.AsyncClient() as client:
        response = await client.post(webhook_url, json=message)
        response.raise_for_status()
    
    print("Report posted to Slack")
```

### Unit Test with Sample Data

```python
import pytest
from app.services.aggregator import CommitAggregator

@pytest.mark.asyncio
async def test_aggregate_commits():
    """Test commit aggregation with sample data."""
    # Load sample commits
    with open('docs/examples/sample-commits.json') as f:
        commits = json.load(f)
    
    # Aggregate
    aggregator = CommitAggregator()
    result = await aggregator.aggregate(commits)
    
    # Assertions
    assert result['total_commits'] == 10
    assert len(result['by_repository']) == 3
    assert 'customer-portal' in result['by_repository']
    assert len(result['by_author']) == 4
```
